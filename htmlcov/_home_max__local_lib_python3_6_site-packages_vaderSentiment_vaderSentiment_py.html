83def negated(input_words, include_nt=True): 

84 """ 

85 Determine if input contains negation words 

86 """ 

87 input_words = [str(w).lower() for w in input_words] 

88 neg_words = [] 

89 neg_words.extend(NEGATE) 

90 for word in neg_words: 

91 if word in input_words: 

92 return True 

93 if include_nt: 

94 for word in input_words: 

95 if "n't" in word: 

96 return True 

97 if "least" in input_words: 

98 i = input_words.index("least") 

99 if i > 0 and input_words[i - 1] != "at": 

100 return True 

101 return False

104def normalize(score, alpha=15): 

105 """ 

106 Normalize the score to be between -1 and 1 using an alpha that 

107 approximates the max expected value 

108 """ 

109 norm_score = score / math.sqrt((score * score) + alpha) 

110 if norm_score < -1.0: 

111 return -1.0 

112 elif norm_score > 1.0: 

113 return 1.0 

114 else: 

115 return norm_score

118def allcap_differential(words): 

119 """ 

120 Check whether just some words in the input are ALL CAPS 

121 :param list words: The words to inspect 

122 :returns: `True` if some but not all items in `words` are ALL CAPS 

123 """ 

124 is_different = False 

125 allcap_words = 0 

126 for word in words: 

127 if word.isupper(): 

128 allcap_words += 1 

129 cap_differential = len(words) - allcap_words 

130 if 0 < cap_differential < len(words): 

131 is_different = True 

132 return is_different

135def scalar_inc_dec(word, valence, is_cap_diff): 

136 """ 

137 Check if the preceding words increase, decrease, or negate/nullify the 

138 valence 

139 """ 

140 scalar = 0.0 

141 word_lower = word.lower() 

142 if word_lower in BOOSTER_DICT: 

143 scalar = BOOSTER_DICT[word_lower] 

144 if valence < 0: 

145 scalar *= -1 

146 # check if booster/dampener word is in ALLCAPS (while others aren't) 

147 if word.isupper() and is_cap_diff: 

148 if valence > 0: 

149 scalar += C_INCR 

150 else: 

151 scalar -= C_INCR 

152 return scalar

155class SentiText(object): 

156 """ 

157 Identify sentiment-relevant string-level properties of input text. 

158 """ 

159 

160 def __init__(self, text): 

161 if not isinstance(text, str): 

162 text = str(text).encode('utf-8') 

163 self.text = text 

164 self.words_and_emoticons = self._words_and_emoticons() 

165 # doesn't separate words from\ 

166 # adjacent punctuation (keeps emoticons & contractions) 

167 self.is_cap_diff = allcap_differential(self.words_and_emoticons)

169 def _words_plus_punc(self): 

170 """ 

171 Returns mapping of form: 

172 { 

173 'cat,': 'cat', 

174 ',cat': 'cat', 

175 } 

176 """ 

177 no_punc_text = REGEX_REMOVE_PUNCTUATION.sub('', self.text) 

178 # removes punctuation (but loses emoticons & contractions) 

179 words_only = no_punc_text.split() 

180 # remove singletons 

181 words_only = set(w for w in words_only if len(w) > 1) 

182 # the product gives ('cat', ',') and (',', 'cat') 

183 punc_before = {''.join(p): p[1] for p in product(PUNC_LIST, words_only)} 

184 punc_after = {''.join(p): p[0] for p in product(words_only, PUNC_LIST)} 

185 words_punc_dict = punc_before 

186 words_punc_dict.update(punc_after) 

187 return words_punc_dict 

188 

189 def _words_and_emoticons(self): 

190 """ 

191 Removes leading and trailing puncutation 

192 Leaves contractions and most emoticons 

193 Does not preserve punc-plus-letter emoticons (e.g. :D) 

194 """ 

195 wes = self.text.split() 

196 words_punc_dict = self._words_plus_punc() 

197 wes = [we for we in wes if len(we) > 1] 

198 for i, we in enumerate(wes): 

199 if we in words_punc_dict: 

200 wes[i] = words_punc_dict[we] 

201 return wes

241 def polarity_scores(self, text): 

242 """ 

243 Return a float for sentiment strength based on the input text. 

244 Positive values are positive valence, negative value are negative 

245 valence. 

246 """ 

247 # convert emojis to their textual descriptions 

248 text_token_list = text.split() 

249 text_no_emoji_lst = [] 

250 for token in text_token_list: 

251 if token in self.emojis: 

252 # get the textual description 

253 description = self.emojis[token] 

254 text_no_emoji_lst.append(description) 

255 else: 

256 text_no_emoji_lst.append(token) 

257 text = " ".join(x for x in text_no_emoji_lst) 

258 

259 sentitext = SentiText(text) 

260 

261 sentiments = [] 

262 words_and_emoticons = sentitext.words_and_emoticons 

263 for item in words_and_emoticons: 

264 valence = 0 

265 i = words_and_emoticons.index(item) 

266 # check for vader_lexicon words that may be used as modifiers or negations 

267 if item.lower() in BOOSTER_DICT: 

268 sentiments.append(valence) 

269 continue 

270 if (i < len(words_and_emoticons) - 1 and item.lower() == "kind" and 

271 words_and_emoticons[i + 1].lower() == "of"): 

272 sentiments.append(valence) 

273 continue 

274 

275 sentiments = self.sentiment_valence(valence, sentitext, item, i, sentiments) 

276 

277 sentiments = self._but_check(words_and_emoticons, sentiments) 

278 

279 valence_dict = self.score_valence(sentiments, text) 

280 

281 return valence_dict

283 def sentiment_valence(self, valence, sentitext, item, i, sentiments): 

284 is_cap_diff = sentitext.is_cap_diff 

285 words_and_emoticons = sentitext.words_and_emoticons 

286 item_lowercase = item.lower() 

287 if item_lowercase in self.lexicon: 

288 # get the sentiment valence 

289 valence = self.lexicon[item_lowercase] 

290 # check if sentiment laden word is in ALL CAPS (while others aren't) 

291 if item.isupper() and is_cap_diff: 

292 if valence > 0: 

293 valence += C_INCR 

294 else: 

295 valence -= C_INCR 

296 

297 for start_i in range(0, 3): 

298 # dampen the scalar modifier of preceding words and emoticons 

299 # (excluding the ones that immediately preceed the item) based 

300 # on their distance from the current item. 

301 if i > start_i and words_and_emoticons[i - (start_i + 1)].lower() not in self.lexicon: 

302 s = scalar_inc_dec(words_and_emoticons[i - (start_i + 1)], valence, is_cap_diff) 

303 if start_i == 1 and s != 0: 

304 s = s * 0.95 

305 if start_i == 2 and s != 0: 

306 s = s * 0.9 

307 valence = valence + s 

308 valence = self._negation_check(valence, words_and_emoticons, start_i, i) 

309 if start_i == 2: 

310 valence = self._special_idioms_check(valence, words_and_emoticons, i) 

311 

312 valence = self._least_check(valence, words_and_emoticons, i) 

313 sentiments.append(valence) 

314 return sentiments

316 def _least_check(self, valence, words_and_emoticons, i): 

317 # check for negation case using "least" 

318 if i > 1 and words_and_emoticons[i - 1].lower() not in self.lexicon \ 

319 and words_and_emoticons[i - 1].lower() == "least": 

320 if words_and_emoticons[i - 2].lower() != "at" and words_and_emoticons[i - 2].lower() != "very": 

321 valence = valence * N_SCALAR 

322 elif i > 0 and words_and_emoticons[i - 1].lower() not in self.lexicon \ 

323 and words_and_emoticons[i - 1].lower() == "least": 

324 valence = valence * N_SCALAR 

325 return valence

433 # check for added emphasis resulting from exclamation points (up to 4 of them) 

434 ep_count = text.count("!") 

435 if ep_count > 4: 

436 ep_count = 4 

437 # (empirically derived mean sentiment intensity rating increase for 

438 # exclamation points) 

439 ep_amplifier = ep_count * 0.292 

440 return ep_amplifier

442 @staticmethod 

443 def _amplify_qm(text): 

444 # check for added emphasis resulting from question marks (2 or 3+) 

445 qm_count = text.count("?") 

446 qm_amplifier = 0 

447 if qm_count > 1: 

448 if qm_count <= 3: 

449 # (empirically derived mean sentiment intensity rating increase for 

450 # question marks) 

451 qm_amplifier = qm_count * 0.18 

452 else: 

453 qm_amplifier = 0.96 

454 return qm_amplifier

456 @staticmethod 

457 def _sift_sentiment_scores(sentiments): 

458 # want separate positive versus negative sentiment scores 

459 pos_sum = 0.0 

460 neg_sum = 0.0 

461 neu_count = 0 

462 for sentiment_score in sentiments: 

463 if sentiment_score > 0: 

464 pos_sum += (float(sentiment_score) + 1) # compensates for neutral words that are counted as 1 

465 if sentiment_score < 0: 

466 neg_sum += (float(sentiment_score) - 1) # when used with math.fabs(), compensates for neutrals 

467 if sentiment_score == 0: 

468 neu_count += 1 

469 return pos_sum, neg_sum, neu_count

471 def score_valence(self, sentiments, text): 

472 if sentiments: 

473 sum_s = float(sum(sentiments)) 

474 # compute and add emphasis from punctuation in text 

475 punct_emph_amplifier = self._punctuation_emphasis(text) 

476 if sum_s > 0: 

477 sum_s += punct_emph_amplifier 

478 elif sum_s < 0: 

479 sum_s -= punct_emph_amplifier 

480 

481 compound = normalize(sum_s) 

482 # discriminate between positive, negative and neutral sentiment scores 

483 pos_sum, neg_sum, neu_count = self._sift_sentiment_scores(sentiments) 

484 

485 if pos_sum > math.fabs(neg_sum): 

486 pos_sum += punct_emph_amplifier 

487 elif pos_sum < math.fabs(neg_sum): 

488 neg_sum -= punct_emph_amplifier 

489 

490 total = pos_sum + math.fabs(neg_sum) + neu_count 

491 pos = math.fabs(pos_sum / total) 

492 neg = math.fabs(neg_sum / total) 

493 neu = math.fabs(neu_count / total) 

494 

495 else: 

496 compound = 0.0 

497 pos = 0.0 

498 neg = 0.0 

499 neu = 0.0 

500 

501 sentiment_dict = \ 

502 {"neg": round(neg, 3), 

503 "neu": round(neu, 3), 

504 "pos": round(pos, 3), 

505 "compound": round(compound, 4)} 

506 

507 return sentiment_dict
