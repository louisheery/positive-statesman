12def salted_hmac(key_salt, value, secret=None): 

13 """ 

14 Return the HMAC-SHA1 of 'value', using a key generated from key_salt and a 

15 secret (which defaults to settings.SECRET_KEY). 

16 

17 A different key_salt should be passed in for every application of HMAC. 

18 """ 

19 if secret is None: 

20 secret = settings.SECRET_KEY 

21 

22 key_salt = force_bytes(key_salt) 

23 secret = force_bytes(secret) 

24 

25 # We need to generate a derived key from our base key. We can do this by 

26 # passing the key_salt and our base key through a pseudo-random function and 

27 # SHA1 works nicely. 

28 key = hashlib.sha1(key_salt + secret).digest() 

29 

30 # If len(key_salt + secret) > sha_constructor().block_size, the above 

31 # line is redundant and could be replaced by key = key_salt + secret, since 

32 # the hmac module does the same thing for keys longer than the block size. 

33 # However, we need to ensure that we *always* do this. 

34 return hmac.new(key, msg=force_bytes(value), digestmod=hashlib.sha1)

37def get_random_string(length=12, 

38 allowed_chars='abcdefghijklmnopqrstuvwxyz' 

39 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'): 

40 """ 

41 Return a securely generated random string. 

42 

43 The default length of 12 with the a-z, A-Z, 0-9 character set returns 

44 a 71-bit value. log_2((26+26+10)^12) =~ 71 bits 

45 """ 

46 return ''.join(secrets.choice(allowed_chars) for i in range(length))
