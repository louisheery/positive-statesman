17def load_ssh_public_key(data, backend): 

18 key_parts = data.split(b' ', 2) 

19 

20 if len(key_parts) < 2: 

21 raise ValueError( 

22 'Key is not in the proper format or contains extra data.') 

23 

24 key_type = key_parts[0] 

25 

26 if key_type == b'ssh-rsa': 

27 loader = _load_ssh_rsa_public_key 

28 elif key_type == b'ssh-dss': 

29 loader = _load_ssh_dss_public_key 

30 elif key_type in [ 

31 b'ecdsa-sha2-nistp256', b'ecdsa-sha2-nistp384', b'ecdsa-sha2-nistp521', 

32 ]: 

33 loader = _load_ssh_ecdsa_public_key 

34 elif key_type == b'ssh-ed25519': 

35 loader = _load_ssh_ed25519_public_key 

36 else: 

37 raise UnsupportedAlgorithm('Key type is not supported.') 

38 

39 key_body = key_parts[1] 

40 

41 try: 

42 decoded_data = base64.b64decode(key_body) 

43 except TypeError: 

44 raise ValueError('Key is not in the proper format.') 

45 

46 inner_key_type, rest = _ssh_read_next_string(decoded_data) 

47 

48 if inner_key_type != key_type: 

49 raise ValueError( 

50 'Key header and key body contain different key type values.' 

51 ) 

52 

53 return loader(key_type, rest, backend)

116def _ssh_read_next_string(data): 

117 """ 

118 Retrieves the next RFC 4251 string value from the data. 

119 

120 While the RFC calls these strings, in Python they are bytes objects. 

121 """ 

122 if len(data) < 4: 

123 raise ValueError("Key is not in the proper format") 

124 

125 str_len, = struct.unpack('>I', data[:4]) 

126 if len(data) < str_len + 4: 

127 raise ValueError("Key is not in the proper format") 

128 

129 return data[4:4 + str_len], data[4 + str_len:]
